```clojure
;; Good: Manage trading sessions
(defn run-trading-session [symbols duration-minutes]
  (let [start-time (System/currentTimeMillis)
        end-time (+ start-time (* duration-minutes 60000))]
    (loop [signals []]
      (if (< (System/currentTimeMillis) end-time)
        (let [new-signals (map #(generate-signal % highway bowtie config) symbols)]
          (recur (into signals new-signals)))
```
